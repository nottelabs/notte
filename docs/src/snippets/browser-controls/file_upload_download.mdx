```python file_upload_download.py
from notte_sdk import NotteClient, actions

client = NotteClient()

with client.Session() as session:
    session.execute(actions.Goto(url="https://example.com/upload"))

    # Upload file
    session.execute(actions.UploadFile(
        selector="input[type='file']",
        file_path="/path/to/document.pdf"
    ))
    session.execute(actions.Click(selector="button.submit"))

    # Wait for processing
    session.execute(actions.Wait(duration=3))

    # Download result
    session.execute(actions.DownloadFile(selector="a.download"))

    # Access downloaded files
    downloaded_files = session.storage.list_downloaded_files()
    for file in downloaded_files:
        print(f"Downloaded: {file.file_id}")
```
