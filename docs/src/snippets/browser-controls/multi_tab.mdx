```python multi_tab.py
from notte_sdk import NotteClient, actions

client = NotteClient()

def process_data(data):
    print(f"Processed: {data[:100]}...")

with client.Session() as session:
    # Open multiple tabs
    session.execute(actions.Goto(url="https://site1.com"))
    session.execute(actions.GotoNewTab(url="https://site2.com"))
    session.execute(actions.GotoNewTab(url="https://site3.com"))

    # Extract from each tab
    for tab_index in range(3):
        session.execute(actions.SwitchTab(tab_index=tab_index))
        data = session.scrape()
        process_data(data)
```
