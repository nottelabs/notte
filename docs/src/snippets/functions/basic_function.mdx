from notte_sdk import NotteClient

def run(url: str, search_query: str):
    """
    Search a website and extract results.

    Args:
        url: The website URL
        search_query: Search term
    """
    client = NotteClient()

    with client.Session() as session:
        # Navigate to site
        session.execute(type="goto", url=url)

        # Search
        session.execute(type="fill", placeholder="Search", value=search_query)
        session.execute(type="press_key", key="Enter")

        # Extract results
        results = session.scrape(
            instructions="Extract search results"
        )

        return results
