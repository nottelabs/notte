```python agent_fallback.py
from notte_sdk import NotteClient, actions

client = NotteClient()

with client.Session() as session:
    with client.AgentFallback(session, task="Add item to cart") as fb:
        # Try deterministic actions
        session.execute(actions.Click(selector="#add-to-cart"))
        session.execute(actions.Click(selector="#checkout"))

        # If any action fails, agent takes over automatically

    if fb.success:
        print("Task completed!")
    if fb.agent_invoked:
        print("Agent helped recover from failure")
```
