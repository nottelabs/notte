from notte_sdk import NotteClient

notte = NotteClient()
storage = notte.FileStorage()

# Upload files before agent execution
upload_response = storage.upload("/path/to/document.pdf")
print(f"Upload success: {upload_response.success}")

# Create session with storage attached
with notte.Session(storage=storage) as session:
    agent = notte.Agent(session=session, max_steps=5)

    # Agent can now access uploaded files
    response = agent.run(
        task="Process the uploaded PDF document",
        url="https://example.com"
    )

    print(f"Agent task completed: {response.success}")

# List and download files created by the agent
downloaded_files = storage.list(type="downloads")
for file_name in downloaded_files:
    success = storage.download(file_name=file_name, local_dir="/tmp/results")
    print(f"Downloaded {file_name}: {success}")
